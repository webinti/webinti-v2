    // Hero ember scroll intensifier
    (function () {
      const hero = document.querySelector<HTMLElement>('.hero');
      if (!hero) return;
      let ticking = false;
      function update() {
        const r = hero!.getBoundingClientRect();
        const vh = window.innerHeight;
        const p = Math.min(1, Math.max(0, 1 - (r.bottom / (vh + r.height * 0.5))));
        const op = 0.5 + p * 0.45;
        const sc = 1 + p * 0.2;
        hero!.style.setProperty('--ember-opacity', op.toFixed(3));
        hero!.style.setProperty('--ember-scale', sc.toFixed(3));
        ticking = false;
      }
      window.addEventListener('scroll', () => {
        if (!ticking) { requestAnimationFrame(update); ticking = true; }
      }, { passive: true });
      update();
    })();
